<script>

const ctx = document.getElementById('moodChart').getContext('2d');

const moodChart = new Chart(ctx, {

  type: 'line',

  data: {

    labels: ['週一', '週二', '週三', '週四', '週五', '週六', '週日'],

    datasets: [{

      label: '心情分數',

      data: [null, null, null, null, null, null, null],

      borderColor: '#4A7C59',

      backgroundColor: 'rgba(74, 124, 89, 0.1)',

      fill: true,

      tension: 0.3

    }]

  },

  options: { scales: { y: { min: 0, max: 10, title: { display: true, text: '心情分數' } } } }

});


function updateChart() {

  const values = [];

  for (let i = 1; i <= 7; i++) {

    const val = parseFloat(document.getElementById(`day${i}`).value);

    values.push(isNaN(val) ? null : val);

  }

  moodChart.data.datasets[0].data = values;

  moodChart.update();

}


function saveChart() {

  const link = document.createElement('a');

  link.download = '我的心情變化圖.png';

  link.href = document.getElementById('moodChart').toDataURL('image/png');

  link.click();

}

</script>
